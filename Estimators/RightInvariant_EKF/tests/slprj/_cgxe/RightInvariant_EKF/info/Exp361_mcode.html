1function [ output ] = Exp( v )
 2%EXP Computes the vectorized exponential map (at the identity) as defined in:
 3% http://www.roboticsproceedings.org/rss11/p06.pdf
 4
 5A = Angles.skew(v);
 6norm_v = norm(v);
 7
 8if norm_v == 0
 9    output = eye(3);
10else
11    output = eye(3) + (sin(norm_v)/norm_v)*A + ((1-cos(norm_v))/(norm_v^2))*A^2;
12end
13
14end
