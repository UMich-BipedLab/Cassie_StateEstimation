108        function setupImpl(obj)

110            obj.params = CassieParameters;
111            obj.filter_enabled = false;
112            obj.bias_initialized = false;
113            obj.a_init_vec = zeros(3, obj.imu_init_total_count);
114            obj.w_init_vec = zeros(3, obj.imu_init_total_count);
115            
116            % Initialize Sensor Covariances
117            obj.Qg = diag(obj.gyro_noise_std.^2);          % Gyro Covariance Matrix
118            obj.Qbg = diag(obj.gyro_bias_noise_std.^2);    % Gyro bias Covariance Matrix
119            obj.Qa = diag(obj.accel_noise_std.^2);         % Accel Covariance Matrix
120            obj.Qba = diag(obj.accel_bias_noise_std.^2);   % Accel Bias Covariance Matrix
121            obj.Qc = diag(obj.contact_noise_std.^2);       % Contact Covariance Matrix
122            obj.Qe = diag(obj.encoder_noise_std.^2);       % Encoder Covariance Matrix
123            obj.Ql = diag(obj.landmark_noise_std.^2);      % Landmark Covariance Matrix
124            obj.Np = diag(obj.prior_forward_kinematics_std.^2); % Prior Forward Kinematics Covariance Matrix
125            obj.P_prior = blkdiag(diag(obj.prior_base_pose_std(1:3).^2), ...
126                                  diag(obj.prior_base_velocity_std.^2), ...
127                                  diag(obj.prior_base_pose_std(4:6).^2), ...
128                                  diag(obj.prior_contact_position_std.^2), ...
129                                  diag(obj.prior_contact_position_std.^2), ...
130                                  diag(obj.prior_gyro_bias_std.^2),...
131                                  diag(obj.prior_accel_bias_std.^2));
132                                         
133            % Initialize bias estimates
134            obj.bg0 = obj.gyro_bias_init;
135            obj.ba0 = obj.accel_bias_init;

137            % Initialze State and Covariance

141            obj.P = eye(21);

162        function [X, theta, P, enabled, landmark_ids] = stepImpl(obj, enable, t, w, a, encoders, contact, measured_landmarks, X_init)
163            % Function that creates a state vector from sensor readings.
164            %
165            %   Inputs:
166            %       enable    - flag to enable/disable the filter
167            %       t         - time
168            %       w         - angular velocity, {I}_w_{WI}
169            %       a         - linear acceleration, {I}_a_{WI}
170            %       encoders  - joint encoder positions
171            %       contact   - contact indicator
172            %       measured_landmarks - vector of landmark IDs along with distance
173            %       X_init    - initial state
174            %
175            %   Outputs:
176            %       X       - current state estimate
177            %       theta   - current parameter estimates
178            %       P       - current covariance estimate 
179            %       enabled - flag indicating when the filter is enabled
180            %
181            %   Author: Ross Hartley
182            %   Date:   1/19/2018
183            %
184            
185            % Initialize bias
186            % (does nothing if bias is already initialized)
187            obj.InitializeBias(w, a, X_init)
188            
189            % Initiaze filter
190            % (does nothing if filter is already initialized)
191            if t > 0.01 && any(contact == 1) 
192                obj.InitializeFilter(enable, X_init);
193            end
194            
195            % Only run if filter is enabled
196            if obj.filter_enabled
197           
198                % Predict state using IMU and contact measurements     
199                obj.Predict_State(obj.w_prev, obj.a_prev, obj.encoders_prev, obj.contact_prev, t - obj.t_prev);
200
201                % Update using other measurements
202                if obj.ekf_update_enabled
203                    % Update state using forward kinematic measurements
204                    obj.Update_ForwardKinematics(encoders, contact);
205                    % Update state using landmark position measurements
206                    if ~isempty(measured_landmarks)
207                        if obj.enable_static_landmarks
208                            obj.Update_StaticLandmarks(measured_landmarks);
209                        else
210                            obj.Update_Landmarks(measured_landmarks);
211                        end
212                    end
213                end
214            
215            end
216            
217            % Store last values
218            obj.w_prev = w;
219            obj.a_prev = a;
220            obj.encoders_prev = encoders;
221            obj.contact_prev = contact;
222            obj.t_prev = t;
223           
224            % Output
225            X = obj.X;
226            theta = obj.theta;
227            P = obj.P(1:21,1:21);
228            enabled = double(obj.filter_enabled);
229            landmark_ids = obj.landmark_ids;
230
231        end % stepImpl

279        function [] = Store_Covariance(obj, Pa)
280            % Separate covariance into state, landmark, and parameter parts
281            assert(size(Pa,2) == 21 + 3*obj.num_landmarks);            
282            obj.Pxx = Pa(1:15, 1:15);
283            obj.Pll(1:3*obj.num_landmarks, 1:3*obj.num_landmarks) = Pa(16:16+3*obj.num_landmarks-1, 16:16+3*obj.num_landmarks-1);
284            obj.Pxl(1:15, 1:3*obj.num_landmarks) = Pa(1:15, 16:16+3*obj.num_landmarks-1);
285            obj.Ptt = Pa(end-5:end, end-5:end);
286            obj.Pxt = Pa(1:15, end-5:end);
287            obj.Plt(1:3*obj.num_landmarks,:) = Pa(16:16+3*obj.num_landmarks-1, end-5:end);
288        end

294                  obj.Pxt', obj.Plt(1:3*obj.num_landmarks,:)', obj.Ptt];

302        function [R, v, p, dR, dL, bg, ba] = Separate_State(~, X, theta)
303            % Separate state vector into components
304            R = X(1:3,1:3);  % Orientation
305            v = X(1:3,4);    % Base Velocity
306            p = X(1:3,5);    % Base Position
307            dR = X(1:3,6);   % Right Foot Position
308            dL = X(1:3,7);   % Left Foot Position
309            
310            % Parameters
311            bg = theta(1:3); % Gyroscope Bias
312            ba = theta(4:6); % Accelerometer Bias
313        end

322        function [A] = skew(~, v)
323            % Convert from vector to skew symmetric matrix
324            A = [    0, -v(3),  v(2);
325                  v(3),     0, -v(1);
326                  -v(2),  v(1),   0];
327        end

329        function [dX, dtheta] = exp(obj, v)
330            % Exponential map of SE_N(3)
331            A = obj.skew(v(1:3));
332            phi = norm(v(1:3));
333            if phi < 1e-6
334                R = eye(3);
335                Jr = eye(3);
336            else
337                R = eye(3) + (sin(phi)/phi)*A + ((1-cos(phi))/(phi^2))*A^2;
338                Jr = eye(3) + ((1-cos(phi))/(phi^2))*A + ((phi-sin(phi))/(phi^3))*A*A;
339            end
340            u = reshape(v(4:end),3,[]);
341            
342            dX = eye(3 + size(u,2));
343            dX(1:3,1:3) = R;
344            dX(1:3,4:end) = Jr*u;
345            
346            dtheta = v(end-5:end);
347        end

349        function AdjX = Adjoint(obj, X)
350            % Adjoint of SE_N(3)         
351            N = size(X,2)-3;
352            R = X(1:3,1:3);
353            AdjX = zeros(3*(N+1));
354            AdjX(1:3,1:3) = R;
355            for i = 1:N
356                AdjX(3*i+1:3*i+3,3*i+1:3*i+3) = R;
357                AdjX(3*i+1:3*i+3,1:3) = obj.skew(X(1:3,i+3))*R;
358            end
359        end

361        function [] = InitializeBias(obj, w, a, X_init)
362            % Function to initiaze IMU bias
363            if obj.static_bias_initialization
364                if ~obj.bias_initialized
365                    if norm(a) > 0 && norm(w) > 0 % Wait for valid signal
366                        if obj.imu_init_count <= obj.imu_init_total_count
367                            % Accumulate vector of imu data
368                            Rwi = X_init(1:3,1:3);
369                            obj.a_init_vec(:,obj.imu_init_count) = Rwi'*(Rwi*a + obj.g);
370                            obj.w_init_vec(:,obj.imu_init_count) = w;
371                            obj.imu_init_count = obj.imu_init_count + 1;
372                        else
373                            % Use imu data vector to estimate initial biases
374                            obj.ba0 = mean(obj.a_init_vec,2);
375                            obj.bg0 = mean(obj.w_init_vec,2);
376                            obj.bias_initialized = true;
377                        end
378                    end
379                end
380            else
381                obj.bias_initialized = true;
382            end
383        end

385        function [] = InitializeFilter(obj, enable, X_init)
386            % Attempt to enable filter (successful if enable is true, and
387            % at least one foot is on the ground)
388            if enable && ~obj.filter_enabled
389                obj.X = X_init;
390                obj.theta = [obj.bg0; obj.ba0];
391                obj.Store_Covariance(obj.P_prior);
392                obj.filter_enabled = true;
393            end
394
395            % If filter is disabled, zero everything
396            if ~enable || ~obj.filter_enabled
397                obj.X = eye(7);
398                obj.landmarks = zeros(3,obj.max_landmarks);
399                obj.theta = zeros(6,1);
400                obj.Store_Covariance(eye(21));
401                obj.num_landmarks = 0;
402                obj.landmark_ids = zeros(1,obj.max_landmarks);
403                obj.filter_enabled = false;
404            end
405            
406        end

408        function [] = Predict_State(obj, w, a, encoders, contact, dt)
409            % Right-Invariant Extended Kalman Filter Prediction Step     
410            [R, v, p, dR, dL, bg, ba] = obj.Separate_State(obj.X, obj.theta);
411            
412            % Bias corrected IMU information
413            w_k = w - bg;    % {I}_w_{WI}
414            a_k = a - ba;    % {I}_a_{WI}
415            
416            % Base Pose Dynamics
417            R_pred = R * Angles.Exp(w_k*dt);
418            v_pred = v + (R*a_k + obj.g)*dt;
419            p_pred = p + v*dt + 0.5*(R*a_k + obj.g)*dt^2;
420            
421            % Foot Position Dynamics
422            dR_off = p_pred + R_pred * p_VectorNav_to_RightToeBottom(encoders); % {W}_p_{WR}
423            dL_off = p_pred + R_pred * p_VectorNav_to_LeftToeBottom(encoders);  % {W}_p_{WL}
424            dR_pred = contact(2)*dR + (1-contact(2))*dR_off;
425            dL_pred = contact(1)*dL + (1-contact(1))*dL_off;
426           
427            % Bias Dynamics
428            bg_pred = bg; 
429            ba_pred = ba;
430                        
431            
432            % -- Linearized invariant error dynamics --
433            
434            % Base
435            Fc = [       zeros(3), zeros(3), zeros(3), zeros(3), zeros(3);
436                  obj.skew(obj.g), zeros(3), zeros(3), zeros(3), zeros(3);
437                         zeros(3),   eye(3), zeros(3), zeros(3), zeros(3);
438                         zeros(3), zeros(3), zeros(3), zeros(3), zeros(3);
439                         zeros(3), zeros(3), zeros(3), zeros(3), zeros(3)];
440           % Landmarks          
441           Fc = blkdiag(Fc, obj.Zeros(1:3*obj.num_landmarks,1:3*obj.num_landmarks));    
442           
443           % Parameters
444           Fc = blkdiag(Fc, zeros(6));               
445           Fc(1:15,end-5:end) = [             -R, zeros(3);
446                                  -obj.skew(v)*R,       -R;
447                                  -obj.skew(p)*R, zeros(3);
448                                 -obj.skew(dR)*R, zeros(3);
449                                 -obj.skew(dL)*R, zeros(3)];
450           
451            % Discretize
452            Fk = eye(size(Fc)) + Fc*dt; 
453            
454            Lc = blkdiag(obj.Adjoint(obj.Augmented_State()), eye(6));
455            hR_R = R_VectorNav_to_RightToeBottom(encoders);
456            hR_L = R_VectorNav_to_LeftToeBottom(encoders);
457            Q = blkdiag(obj.Qg, obj.Qa, zeros(3), hR_R*(obj.Qc+(1e4*eye(3).*(1-contact(2))))*hR_R', hR_L*(obj.Qc+(1e4*eye(3).*(1-contact(1))))*hR_L', obj.Zeros(1:3*obj.num_landmarks,1:3*obj.num_landmarks), obj.Qbg, obj.Qba);             
458            Qk = Fk*Lc*Q*Lc'*Fk'*dt; % Discretized
459            
460            % Construct predicted state
461            [obj.X, obj.theta] = obj.Construct_State(R_pred, v_pred, p_pred, dR_pred, dL_pred, bg_pred, ba_pred);
462            
463            % Predict Covariance
464            obj.Store_Covariance( Fk * obj.Augmented_Covariance() * Fk' + Qk );
465        end

467        function [] = Update_State(obj, Y, b, H, N, PI)
468            % Update State and Covariance from a measurement
469            % Compute Kalman Gain
470            Xa = obj.Augmented_State();
471            Pa = obj.Augmented_Covariance();
472            S = H * Pa * H' + N;
473            K = (Pa * H')/S;
474            
475            % Copy X along the diagonals if more than one measurement
476            % TODO: find better way to do this
477            Xdim = 7 + obj.num_landmarks;
478            Ydim = length(Y)/Xdim;
479            assert(Ydim < obj.max_landmarks);
480            assert(Xdim < 7 + obj.max_landmarks);
481            coder.varsize('BigX', [7+obj.max_landmarks, 7+obj.max_landmarks], [1 1]);
482            BigX = obj.Zeros(1:Xdim*Ydim, 1:Xdim*Ydim);
483            for i = 1:Ydim
484                BigX(Xdim*(i-1)+1:Xdim*(i-1)+Xdim, Xdim*(i-1)+1:Xdim*(i-1)+Xdim) = Xa;
485            end
486            Z = BigX * Y - b;
487            
488            % Update State
489            coder.varsize('delta', [21+3*obj.max_landmarks, 1], [1 0]);
490            delta = K*PI*Z;
491            dX = obj.exp(delta(1:end-6));
492            dtheta = delta(end-5:end);
493            Xa = dX * Xa;
494            obj.X = Xa(1:7,1:7);
495            obj.landmarks(:,1:obj.num_landmarks) = Xa(1:3,8:8+obj.num_landmarks-1);
496            obj.theta = obj.theta + dtheta;
497            
498            % Update Covariance
499            I = eye(size(Pa));
500            obj.Store_Covariance( (I - K*H)* Pa *(I - K*H)' + K*N*K' ); % Joseph update form
501            
502        end

504        function [] = Update_ForwardKinematics(obj, encoders, contact)
505            % Function to perform Right-Invariant EKF update from forward
506            % kinematic measurements
507            
508            R_pred = obj.X(1:3,1:3);
509            
510            if contact(2) == 1 && contact(1) == 1 
511                % Double Support
512                s_pR = p_VectorNav_to_RightToeBottom(encoders);
513                s_pL = p_VectorNav_to_LeftToeBottom(encoders);
514                JR = J_VectorNav_to_RightToeBottom(encoders);
515                JL = J_VectorNav_to_LeftToeBottom(encoders);
516                
517                % Measurement Model
518                Y = [s_pR; 0; 1; -1; 0; obj.Zeros(1:obj.num_landmarks,1); 
519                     s_pL; 0; 1; 0; -1; obj.Zeros(1:obj.num_landmarks,1)];
520                b = zeros(size(Y));
521                H = [zeros(3), zeros(3), -eye(3), eye(3), zeros(3), obj.Zeros(1:3,1:3*obj.num_landmarks), zeros(3,6);
522                     zeros(3), zeros(3), -eye(3), zeros(3), eye(3), obj.Zeros(1:3,1:3*obj.num_landmarks), zeros(3,6)];
523                N = blkdiag(R_pred * JR * obj.Qe * JR' * R_pred' + obj.Np, ...
524                            R_pred * JL * obj.Qe * JL' * R_pred' + obj.Np);  
525                PI = [eye(3), zeros(3,4), obj.Zeros(1:3,1:obj.num_landmarks), zeros(3,7), obj.Zeros(1:3,1:obj.num_landmarks);
526                     zeros(3,7), obj.Zeros(1:3,1:obj.num_landmarks), eye(3), zeros(3,4), obj.Zeros(1:3,1:obj.num_landmarks)];
527
528                % Update State
529                obj.Update_State(Y, b, H, N, PI);
530                        
531            elseif contact(2) == 1 
532                % Single Support Right    
533                s_pR = p_VectorNav_to_RightToeBottom(encoders);
534                JR = J_VectorNav_to_RightToeBottom(encoders);
535                
536                % Measurement Model
537                Y = [s_pR; 0; 1; -1; 0; obj.Zeros(1:obj.num_landmarks,1)];
538                b = zeros(size(Y));
539                H = [zeros(3), zeros(3), -eye(3), eye(3), zeros(3), obj.Zeros(1:3,1:3*obj.num_landmarks), zeros(3,6)];
540                N = R_pred * JR * obj.Qe * JR' * R_pred' + obj.Np;
541                PI = [eye(3), zeros(3,4), obj.Zeros(1:3,1:obj.num_landmarks)];
542
543                % Update State
544                obj.Update_State(Y, b, H, N, PI);
545                
546            elseif contact(1) == 1 
547                % Single Support Left
548                s_pL = p_VectorNav_to_LeftToeBottom(encoders);
549                JL = J_VectorNav_to_LeftToeBottom(encoders);
550
551                % Measurement Model
552                Y = [s_pL; 0; 1; 0; -1; obj.Zeros(1:obj.num_landmarks,1)];
553                b = zeros(size(Y));
554                H = [zeros(3), zeros(3), -eye(3), zeros(3), eye(3), obj.Zeros(1:3,1:3*obj.num_landmarks), zeros(3,6)];
555                N = R_pred * JL * obj.Qe * JL' * R_pred' + obj.Np;
556                PI = [eye(3), zeros(3,4), obj.Zeros(1:3,1:obj.num_landmarks)];
557
558                % Update State
559                obj.Update_State(Y, b, H, N, PI);
560                
561            end

566            % Function to perform Right-Invariant EKF update from static

607        function [] = Update_Landmarks(obj, measured_landmarks)
608            % Function to perform Right-Invariant EKF update from estimated
609            % landmark distance measurements            
610            [R_pred, ~, ~, ~, ~, ~, ~] = obj.Separate_State(obj.X, obj.theta);
611                
612            % Determine sizes (needed for codegen), (TODO: there is probably a better way to do this)
613            lm_cnt = 0;
614            for i = 1:size(measured_landmarks,2)
615                id = find(obj.landmark_ids == measured_landmarks(1,i),1);
616                if ~isempty(id)
617                    lm_cnt = lm_cnt + 1;
618                end
619            end
620            Xdim = 7 + obj.num_landmarks;
621            assert(lm_cnt <= 20); % TODO: change to constant variable
622            assert(Xdim <= 20); % TODO: change to constant variable
623            Y = zeros(Xdim*lm_cnt,1);
624            H = zeros(3*lm_cnt,3*(Xdim-3)+3+6);
625            N = zeros(3*lm_cnt);
626            PI = zeros(3*lm_cnt,Xdim*lm_cnt);
627            new_landmarks = zeros(4, size(measured_landmarks,2) - lm_cnt);
628            index = 1;
629            
630            % Stack landmark measurements for batch update
631            for i = 1:size(measured_landmarks,2)
632                % Search to see if measured landmark id is in the list of
633                % static landmarks
634                id = find(obj.landmark_ids == measured_landmarks(1,i)); id = id(1);  % makes codegen happy
635                if isempty(id)
636                    new_landmarks(:,index) = measured_landmarks(:,i);
637                    index = index + 1;
638                else
639                    % Create measurement model
640                    Y1 = [measured_landmarks(2:end,i); 0; 1; 0; 0];
641                    Y2 = obj.Zeros(1:obj.num_landmarks,1);
642                    Y2(id) = -1;
643                    Y(Xdim*(i-1)+1:Xdim*(i-1)+Xdim) = [Y1; Y2];
644                    
645                    H1 = [zeros(3), zeros(3), -eye(3), zeros(3), zeros(3)];
646                    H2 = obj.Zeros(1:3,1:3*obj.num_landmarks);
647                    H2(:,3*(id-1)+1:3*(id-1)+3) = eye(3);
648                    H3 = zeros(3,6);
649                    H(3*(i-1)+1:3*(i-1)+3,:) = [H1, H2, H3];
650                    
651                    N(3*(i-1)+1:3*(i-1)+3,3*(i-1)+1:3*(i-1)+3) = R_pred * obj.Ql * R_pred';
652                    PI(3*(i-1)+1:3*(i-1)+3,Xdim*(i-1)+1:Xdim*(i-1)+Xdim) = [eye(3), zeros(3,4), zeros(3,obj.num_landmarks)];
653                end
654            end
655                 
656            % Update State
657            if ~isempty(Y)
658                b = zeros(size(Y));
659                obj.Update_State(Y, b, H, N, PI);
660            end
661            
662            % Augment State for new landmarks
663            if ~isempty(new_landmarks)
664                [R, ~, p, ~, ~, ~, ~] = obj.Separate_State(obj.X, obj.theta);
665                for i = 1:size(new_landmarks,2)
666                    if obj.num_landmarks < obj.max_landmarks
667                        % Increment number of landmarks after extracting cov
668                        Pa = obj.Augmented_Covariance();
669                        obj.num_landmarks = obj.num_landmarks + 1; % this needs to be done before Store_Covariance
670
671                        % Initialize new landmark mean
672                        obj.landmark_ids(obj.num_landmarks) = new_landmarks(1,i);
673                        obj.landmarks(:,obj.num_landmarks) = p + R*new_landmarks(2:end,i);
674
675                        % Initialize new landmark covariance
676                        % Start with I with state dim and add row to increase dimension
677                        F = vertcat(eye(size(Pa,1) - 6), [zeros(3,6), eye(3), zeros(3,6), obj.Zeros(1:3,1:3*(obj.num_landmarks-1))]); 
678                        F = blkdiag(F, eye(6)); % Add block I for parameters
679                        G = zeros(size(F,1),3);
680                        G(end-8:end-6,:) = R;
681                        obj.Store_Covariance( F*Pa*F' + G*obj.Ql*G' );
682                    end
683                end
684            end
685            
686        end
