108        function setupImpl(obj)
109            %SETUPIMPL Initialize System object.
110            obj.params = CassieParameters;
111            obj.filter_enabled = false;
112            obj.bias_initialized = false;
113            obj.a_init_vec = zeros(3, obj.imu_init_total_count);
114            obj.w_init_vec = zeros(3, obj.imu_init_total_count);
115            
116            % Initialize Sensor Covariances
117            obj.Qg = diag(obj.gyro_noise_std.^2);          % Gyro Covariance Matrix
118            obj.Qbg = diag(obj.gyro_bias_noise_std.^2);    % Gyro bias Covariance Matrix
119            obj.Qa = diag(obj.accel_noise_std.^2);         % Accel Covariance Matrix
120            obj.Qba = diag(obj.accel_bias_noise_std.^2);   % Accel Bias Covariance Matrix
121            obj.Qc = diag(obj.contact_noise_std.^2);       % Contact Covariance Matrix
122            obj.Qe = diag(obj.encoder_noise_std.^2);       % Encoder Covariance Matrix
123            obj.Ql = diag(obj.landmark_noise_std.^2);      % Landmark Covariance Matrix
124            obj.Np = diag(obj.prior_forward_kinematics_std.^2); % Prior Forward Kinematics Covariance Matrix
125            obj.P_prior = blkdiag(diag(obj.prior_base_pose_std(1:3).^2), ...
126                                  diag(obj.prior_base_velocity_std.^2), ...
127                                  diag(obj.prior_base_pose_std(4:6).^2), ...
128                                  diag(obj.prior_contact_position_std.^2), ...
129                                  diag(obj.prior_contact_position_std.^2), ...
130                                  diag(obj.prior_gyro_bias_std.^2),...
131                                  diag(obj.prior_accel_bias_std.^2));
132                                         
133            % Initialize bias estimates
134            obj.bg0 = obj.gyro_bias_init;
135            obj.ba0 = obj.accel_bias_init;
136            
137            % Initialze State and Covariance
138            obj.X = eye(7);
139            obj.landmarks = zeros(3,obj.max_landmarks);
140            obj.theta = zeros(6,1);
141            obj.P = eye(21);

162        function [X, theta, P, enabled, landmarks_with_ids] = stepImpl(obj, enable, t, w, a, encoders, contact, measured_landmarks, X_init)
163            % Function that creates a state vector from sensor readings.
164            %
165            %   Inputs:
166            %       enable    - flag to enable/disable the filter
167            %       t         - time
168            %       w         - angular velocity, {I}_w_{WI}
169            %       a         - linear acceleration, {I}_a_{WI}
170            %       encoders  - joint encoder positions
171            %       contact   - contact indicator
172            %       measured_landmarks - vector of landmark IDs along with distance
173            %       X_init    - initial state
174            %
175            %   Outputs:
176            %       X       - current state estimate
177            %       theta   - current parameter estimates
178            %       P       - current covariance estimate 
179            %       enabled - flag indicating when the filter is enabled
180            %
181            %   Author: Ross Hartley
182            %   Date:   1/19/2018
183            %
184            
185            % Initialize bias
186            % (does nothing if bias is already initialized)
187            obj.InitializeBias(w, a, X_init)
188            
189            % Initiaze filter
190            % (does nothing if filter is already initialized)
191            if t > 0.01 && any(contact == 1) 
192                obj.InitializeFilter(enable, X_init);
193            end
194            
195            % Only run if filter is enabled
196            if obj.filter_enabled
197           
198                % Predict state using IMU and contact measurements     
199                obj.Predict_State(obj.w_prev, obj.a_prev, obj.encoders_prev, obj.contact_prev, t - obj.t_prev);
200
201                % Update using other measurements
202                if obj.ekf_update_enabled
203                    % Update state using forward kinematic measurements
204                    obj.Update_ForwardKinematics(encoders, contact);
205                    % Update state using landmark position measurements
206                    if ~all(all(isnan(measured_landmarks))) %~isempty(measured_landmarks)
207                        if obj.enable_static_landmarks
208                            obj.Update_StaticLandmarks(measured_landmarks);
209                        else
210                            obj.Update_Landmarks(measured_landmarks);
211                        end
212                    end
213                end
214            
215            end
216            
217            % Store last values
218            obj.w_prev = w;
219            obj.a_prev = a;
220            obj.encoders_prev = encoders;
221            obj.contact_prev = contact;
222            obj.t_prev = t;
223           
224            % Output
225            X = obj.X;
226            theta = obj.theta;
227            Pa = obj.Augmented_Covariance();
228            P = Pa(1:21,1:21);
229            enabled = double(obj.filter_enabled);
230            landmarks_with_ids = [obj.landmark_ids; obj.landmarks];
231
232        end % stepImpl

280        function [] = Store_Covariance(obj, Pa)
281            % Separate covariance into state, landmark, and parameter parts
282            assert(size(Pa,2) == 21 + 3*obj.num_landmarks);            
283            obj.Pxx = Pa(1:15, 1:15);
284            obj.Pll(1:3*obj.num_landmarks, 1:3*obj.num_landmarks) = Pa(16:16+3*obj.num_landmarks-1, 16:16+3*obj.num_landmarks-1);
285            obj.Pxl(1:15, 1:3*obj.num_landmarks) = Pa(1:15, 16:16+3*obj.num_landmarks-1);
286            obj.Ptt = Pa(end-5:end, end-5:end);
287            obj.Pxt = Pa(1:15, end-5:end);
288            obj.Plt(1:3*obj.num_landmarks,:) = Pa(16:16+3*obj.num_landmarks-1, end-5:end);
289        end

292            % Augmented covariance matrix

295                  obj.Pxt', obj.Plt(1:3*obj.num_landmarks,:)', obj.Ptt];

303        function [R, v, p, dR, dL, bg, ba] = Separate_State(~, X, theta)
304            % Separate state vector into components
305            R = X(1:3,1:3);  % Orientation
306            v = X(1:3,4);    % Base Velocity
307            p = X(1:3,5);    % Base Position
308            dR = X(1:3,6);   % Right Foot Position
309            dL = X(1:3,7);   % Left Foot Position
310            
311            % Parameters
312            bg = theta(1:3); % Gyroscope Bias
313            ba = theta(4:6); % Accelerometer Bias
314        end

323        function [A] = skew(~, v)
324            % Convert from vector to skew symmetric matrix
325            A = [    0, -v(3),  v(2);
326                  v(3),     0, -v(1);
327                  -v(2),  v(1),   0];
328        end

330        function [dX, dtheta] = exp(obj, v)
331            % Exponential map of SE_N(3)
332            A = obj.skew(v(1:3));
333            phi = norm(v(1:3));
334            if phi < 1e-6
335                R = eye(3);
336                Jr = eye(3);
337            else
338                R = eye(3) + (sin(phi)/phi)*A + ((1-cos(phi))/(phi^2))*A^2;
339                Jr = eye(3) + ((1-cos(phi))/(phi^2))*A + ((phi-sin(phi))/(phi^3))*A*A;
340            end
341            u = reshape(v(4:end),3,[]);
342            
343            dX = eye(3 + size(u,2));
344            dX(1:3,1:3) = R;
345            dX(1:3,4:end) = Jr*u;
346            
347            dtheta = v(end-5:end);
348        end

350        function AdjX = Adjoint(obj, X)
351            % Adjoint of SE_N(3)         
352            N = size(X,2)-3;
353            R = X(1:3,1:3);
354            AdjX = zeros(3*(N+1));
355            AdjX(1:3,1:3) = R;
356            for i = 1:N
357                AdjX(3*i+1:3*i+3,3*i+1:3*i+3) = R;
358                AdjX(3*i+1:3*i+3,1:3) = obj.skew(X(1:3,i+3))*R;
359            end
360        end

362        function [] = InitializeBias(obj, w, a, X_init)
363            % Function to initiaze IMU bias
364            if obj.static_bias_initialization
365                if ~obj.bias_initialized
366                    if norm(a) > 0 && norm(w) > 0 % Wait for valid signal
367                        if obj.imu_init_count <= obj.imu_init_total_count
368                            % Accumulate vector of imu data
369                            Rwi = X_init(1:3,1:3);
370                            obj.a_init_vec(:,obj.imu_init_count) = Rwi'*(Rwi*a + obj.g);
371                            obj.w_init_vec(:,obj.imu_init_count) = w;
372                            obj.imu_init_count = obj.imu_init_count + 1;
373                        else
374                            % Use imu data vector to estimate initial biases
375                            obj.ba0 = mean(obj.a_init_vec,2);
376                            obj.bg0 = mean(obj.w_init_vec,2);
377                            obj.bias_initialized = true;
378                        end
379                    end
380                end
381            else
382                obj.bias_initialized = true;
383            end
384        end

386        function [] = InitializeFilter(obj, enable, X_init)
387            % Attempt to enable filter (successful if enable is true, and
388            % at least one foot is on the ground)
389            if enable && ~obj.filter_enabled
390                obj.X = X_init;
391                obj.theta = [obj.bg0; obj.ba0];
392                obj.Store_Covariance(obj.P_prior);
393                obj.filter_enabled = true;
394            end
395
396            % If filter is disabled, zero everything
397            if ~enable || ~obj.filter_enabled
398                obj.X = eye(7);
399                obj.landmarks = zeros(3,obj.max_landmarks);
400                obj.theta = zeros(6,1);
401                obj.Store_Covariance(eye(21));
402                obj.num_landmarks = 0;
403                obj.landmark_ids = zeros(1,obj.max_landmarks);
404                obj.filter_enabled = false;
405            end
406            
407        end

409        function [] = Predict_State(obj, w, a, encoders, contact, dt)
410            % Right-Invariant Extended Kalman Filter Prediction Step     
411            [R, v, p, dR, dL, bg, ba] = obj.Separate_State(obj.X, obj.theta);
412            
413            % Bias corrected IMU information
414            w_k = w - bg;    % {I}_w_{WI}
415            a_k = a - ba;    % {I}_a_{WI}
416            
417            % Base Pose Dynamics
418            R_pred = R * Angles.Exp(w_k*dt);
419            v_pred = v + (R*a_k + obj.g)*dt;
420            p_pred = p + v*dt + 0.5*(R*a_k + obj.g)*dt^2;
421            
422            % Foot Position Dynamics
423            dR_off = p_pred + R_pred * p_VectorNav_to_RightToeBottom(encoders); % {W}_p_{WR}
424            dL_off = p_pred + R_pred * p_VectorNav_to_LeftToeBottom(encoders);  % {W}_p_{WL}
425            dR_pred = contact(2)*dR + (1-contact(2))*dR_off;
426            dL_pred = contact(1)*dL + (1-contact(1))*dL_off;
427           
428            % Bias Dynamics
429            bg_pred = bg; 
430            ba_pred = ba;
431                        
432            
433            % -- Linearized invariant error dynamics --
434            
435            % Base
436            Fc = [       zeros(3), zeros(3), zeros(3), zeros(3), zeros(3);
437                  obj.skew(obj.g), zeros(3), zeros(3), zeros(3), zeros(3);
438                         zeros(3),   eye(3), zeros(3), zeros(3), zeros(3);
439                         zeros(3), zeros(3), zeros(3), zeros(3), zeros(3);
440                         zeros(3), zeros(3), zeros(3), zeros(3), zeros(3)];
441           % Landmarks          
442           Fc = blkdiag(Fc, obj.Zeros(1:3*obj.num_landmarks,1:3*obj.num_landmarks));    
443           
444           % Parameters
445           Fc = blkdiag(Fc, zeros(6));               
446           Fc(1:15,end-5:end) = [             -R, zeros(3);
447                                  -obj.skew(v)*R,       -R;
448                                  -obj.skew(p)*R, zeros(3);
449                                 -obj.skew(dR)*R, zeros(3);
450                                 -obj.skew(dL)*R, zeros(3)];
451           
452            % Discretize
453            Fk = eye(size(Fc)) + Fc*dt; 
454            
455            Lc = blkdiag(obj.Adjoint(obj.Augmented_State()), eye(6));
456            hR_R = R_VectorNav_to_RightToeBottom(encoders);
457            hR_L = R_VectorNav_to_LeftToeBottom(encoders);
458            Q = blkdiag(obj.Qg, obj.Qa, zeros(3), hR_R*(obj.Qc+(1e4*eye(3).*(1-contact(2))))*hR_R', hR_L*(obj.Qc+(1e4*eye(3).*(1-contact(1))))*hR_L', obj.Zeros(1:3*obj.num_landmarks,1:3*obj.num_landmarks), obj.Qbg, obj.Qba);             
459            Qk = Fk*Lc*Q*Lc'*Fk'*dt; % Discretized
460            
461            % Construct predicted state
462            [obj.X, obj.theta] = obj.Construct_State(R_pred, v_pred, p_pred, dR_pred, dL_pred, bg_pred, ba_pred);
463            
464            % Predict Covariance
465            obj.Store_Covariance( Fk * obj.Augmented_Covariance() * Fk' + Qk );
466        end

468        function [] = Update_State(obj, Y, b, H, N, PI)
469            % Update State and Covariance from a measurement
470            % Compute Kalman Gain
471            Xa = obj.Augmented_State();
472            Pa = obj.Augmented_Covariance();
473            S = H * Pa * H' + N;
474            K = (Pa * H')/S;
475            
476            % Copy X along the diagonals if more than one measurement
477            % TODO: find better way to do this
478            Xdim = 7 + obj.num_landmarks;
479            Ydim = length(Y)/Xdim;
480            assert(Ydim < obj.max_landmarks);
481            assert(Xdim < 7 + obj.max_landmarks);
482            coder.varsize('BigX', [obj.max_landmarks*(7+obj.max_landmarks), obj.max_landmarks*(7+obj.max_landmarks)], [1 1]);
483            BigX = obj.Zeros(1:Xdim*Ydim, 1:Xdim*Ydim);
484            for i = 1:Ydim
485                BigX(Xdim*(i-1)+1:Xdim*(i-1)+Xdim, Xdim*(i-1)+1:Xdim*(i-1)+Xdim) = Xa;
486            end
487            Z = BigX * Y - b;
488            
489            % Update State
490            coder.varsize('delta', [21+3*obj.max_landmarks, 1], [1 0]);
491            delta = K*PI*Z;
492            dX = obj.exp(delta(1:end-6));
493            dtheta = delta(end-5:end);
494            Xa = dX * Xa;
495            obj.X = Xa(1:7,1:7);
496            obj.landmarks(:,1:obj.num_landmarks) = Xa(1:3,8:8+obj.num_landmarks-1);
497            obj.theta = obj.theta + dtheta;
498            
499            % Update Covariance
500            I = eye(size(Pa));
501            obj.Store_Covariance( (I - K*H)* Pa *(I - K*H)' + K*N*K' ); % Joseph update form
502            
503        end

505        function [] = Update_ForwardKinematics(obj, encoders, contact)
506            % Function to perform Right-Invariant EKF update from forward
507            % kinematic measurements
508            
509            R_pred = obj.X(1:3,1:3);
510            
511            if contact(2) == 1 && contact(1) == 1 
512                % Double Support
513                s_pR = p_VectorNav_to_RightToeBottom(encoders);
514                s_pL = p_VectorNav_to_LeftToeBottom(encoders);
515                JR = J_VectorNav_to_RightToeBottom(encoders);
516                JL = J_VectorNav_to_LeftToeBottom(encoders);
517                
518                % Measurement Model
519                Y = [s_pR; 0; 1; -1; 0; obj.Zeros(1:obj.num_landmarks,1); 
520                     s_pL; 0; 1; 0; -1; obj.Zeros(1:obj.num_landmarks,1)];
521                b = zeros(size(Y));
522                H = [zeros(3), zeros(3), -eye(3), eye(3), zeros(3), obj.Zeros(1:3,1:3*obj.num_landmarks), zeros(3,6);
523                     zeros(3), zeros(3), -eye(3), zeros(3), eye(3), obj.Zeros(1:3,1:3*obj.num_landmarks), zeros(3,6)];
524                N = blkdiag(R_pred * JR * obj.Qe * JR' * R_pred' + obj.Np, ...
525                            R_pred * JL * obj.Qe * JL' * R_pred' + obj.Np);  
526                PI = [eye(3), zeros(3,4), obj.Zeros(1:3,1:obj.num_landmarks), zeros(3,7), obj.Zeros(1:3,1:obj.num_landmarks);
527                     zeros(3,7), obj.Zeros(1:3,1:obj.num_landmarks), eye(3), zeros(3,4), obj.Zeros(1:3,1:obj.num_landmarks)];
528
529                % Update State
530                obj.Update_State(Y, b, H, N, PI);
531                        
532            elseif contact(2) == 1 
533                % Single Support Right    
534                s_pR = p_VectorNav_to_RightToeBottom(encoders);
535                JR = J_VectorNav_to_RightToeBottom(encoders);
536                
537                % Measurement Model
538                Y = [s_pR; 0; 1; -1; 0; obj.Zeros(1:obj.num_landmarks,1)];
539                b = zeros(size(Y));
540                H = [zeros(3), zeros(3), -eye(3), eye(3), zeros(3), obj.Zeros(1:3,1:3*obj.num_landmarks), zeros(3,6)];
541                N = R_pred * JR * obj.Qe * JR' * R_pred' + obj.Np;
542                PI = [eye(3), zeros(3,4), obj.Zeros(1:3,1:obj.num_landmarks)];
543
544                % Update State
545                obj.Update_State(Y, b, H, N, PI);
546                
547            elseif contact(1) == 1 
548                % Single Support Left
549                s_pL = p_VectorNav_to_LeftToeBottom(encoders);
550                JL = J_VectorNav_to_LeftToeBottom(encoders);
551
552                % Measurement Model
553                Y = [s_pL; 0; 1; 0; -1; obj.Zeros(1:obj.num_landmarks,1)];
554                b = zeros(size(Y));
555                H = [zeros(3), zeros(3), -eye(3), zeros(3), eye(3), obj.Zeros(1:3,1:3*obj.num_landmarks), zeros(3,6)];
556                N = R_pred * JL * obj.Qe * JL' * R_pred' + obj.Np;
557                PI = [eye(3), zeros(3,4), obj.Zeros(1:3,1:obj.num_landmarks)];
558
559                % Update State
560                obj.Update_State(Y, b, H, N, PI);
561                
562            end

566        function [] = Update_StaticLandmarks(obj, measured_landmarks)
567            % Function to perform Right-Invariant EKF update from static
568            % landmark distance measurements
569            
570            [R_pred, ~, ~, ~, ~, ~, ~] = obj.Separate_State(obj.X, obj.theta);
571            
572            % Determine sizes (needed for codegen), (TODO: there is probably a better way to do this)

577                end

579                if ~isempty(id)
580                    lm_cnt = lm_cnt + 1;
581                end
582            end

584            Y = zeros(7*lm_cnt,1);
585            b = zeros(7*lm_cnt,1);
586            H = zeros(3*lm_cnt,21);
587            N = zeros(3*lm_cnt);
588            PI = zeros(3*lm_cnt,7*lm_cnt);
589
590            % Stack landmark measurements (TODO: Optimize this)
591            for i = 1:size(measured_landmarks,2)
592                % Search to see if measured landmark id is in the list of

594                if measured_landmarks(1,i) == 0 || isnan(measured_landmarks(1,i))
595                    continue; % skip if id == 0
596                end
597                id = find(obj.landmark_positions(1,:) == measured_landmarks(1,i),1);

604                    PI(3*(i-1)+1:3*(i-1)+3,7*(i-1)+1:7*(i-1)+7) = [eye(3), zeros(3,4)];
605                end
606            end
607                 
608            % Update State
609            if ~isempty(Y)
610                obj.Update_State(Y, b, H, N, PI);

614        function [] = Update_Landmarks(obj, measured_landmarks)
615            % Function to perform Right-Invariant EKF update from estimated
616            % landmark distance measurements            
617            [R_pred, ~, ~, ~, ~, ~, ~] = obj.Separate_State(obj.X, obj.theta);
618                
619            % Determine sizes (needed for codegen), (TODO: there is probably a better way to do this)
620            lm_cnt = 0;
621            for i = 1:size(measured_landmarks,2)
622                if measured_landmarks(1,i) == 0 || isnan(measured_landmarks(1,i))
623                    continue; % skip if id == 0
624                end
625                id = find(obj.landmark_ids == measured_landmarks(1,i),1);
626                if ~isempty(id)
627                    lm_cnt = lm_cnt + 1;
628                end
629            end
630            Xdim = 7 + obj.num_landmarks;
631            assert(lm_cnt <= 20); % TODO: change to constant variable
632            assert(Xdim <= 20); % TODO: change to constant variable
633            Y = zeros(Xdim*lm_cnt,1);
634            H = zeros(3*lm_cnt,3*(Xdim-3)+3+6);
635            N = zeros(3*lm_cnt);
636            PI = zeros(3*lm_cnt,Xdim*lm_cnt);
637            new_landmarks = zeros(4, sum(measured_landmarks(1,:) > 1) - lm_cnt);
638            index = 1;
639            
640            % Stack landmark measurements for batch update
641            for i = 1:size(measured_landmarks,2)
642                % Search to see if measured landmark id is in the list of
643                % static landmarks 
644                if measured_landmarks(1,i) == 0 || isnan(measured_landmarks(1,i))
645                    continue; % skip if id == 0
646                end
647                id = find(obj.landmark_ids == measured_landmarks(1,i)); 
648                if isempty(id)
649                    new_landmarks(:,index) = measured_landmarks(:,i);
650                    index = index + 1;
651                else
652                    id = id(1);  % makes codegen happy
653                    % Create measurement model
654                    Y1 = [measured_landmarks(2:end,i); 0; 1; 0; 0];
655                    Y2 = obj.Zeros(1:obj.num_landmarks,1);
656                    Y2(id) = -1;
657                    Y(Xdim*(i-1)+1:Xdim*(i-1)+Xdim) = [Y1; Y2];
658                    
659                    H1 = [zeros(3), zeros(3), -eye(3), zeros(3), zeros(3)];
660                    H2 = obj.Zeros(1:3,1:3*obj.num_landmarks);
661                    H2(:,3*(id-1)+1:3*(id-1)+3) = eye(3);
662                    H3 = zeros(3,6);
663                    H(3*(i-1)+1:3*(i-1)+3,:) = [H1, H2, H3];
664                    
665                    N(3*(i-1)+1:3*(i-1)+3,3*(i-1)+1:3*(i-1)+3) = R_pred * obj.Ql * R_pred';
666                    PI(3*(i-1)+1:3*(i-1)+3,Xdim*(i-1)+1:Xdim*(i-1)+Xdim) = [eye(3), zeros(3,4), obj.Zeros(1:3,1:obj.num_landmarks)];
667                end
668            end
669                 
670            % Update State
671            if ~isempty(Y)
672                b = zeros(size(Y));
673                obj.Update_State(Y, b, H, N, PI);
674            end
675            
676            % Augment State for new landmarks
677            if ~isempty(new_landmarks)
678                [R, ~, p, ~, ~, ~, ~] = obj.Separate_State(obj.X, obj.theta);
679                for i = 1:size(new_landmarks,2)
680                    if measured_landmarks(1,i) == 0 || isnan(measured_landmarks(1,i))
681                        continue; % skip if id == 0
682                    end
683                    if obj.num_landmarks < obj.max_landmarks
684                        % Increment number of landmarks after extracting cov
685                        Pa = obj.Augmented_Covariance();
686                        obj.num_landmarks = obj.num_landmarks + 1; % this needs to be done before Store_Covariance
687
688                        % Initialize new landmark mean
689                        obj.landmark_ids(obj.num_landmarks) = new_landmarks(1,i);
690                        obj.landmarks(:,obj.num_landmarks) = p + R*new_landmarks(2:end,i);
691
692                        % Initialize new landmark covariance
693                        % Start with I with state dim and add row to increase dimension
694                        F = vertcat(eye(size(Pa,1) - 6), [zeros(3,6), eye(3), zeros(3,6), obj.Zeros(1:3,1:3*(obj.num_landmarks-1))]); 
695                        F = blkdiag(F, eye(6)); % Add block I for parameters
696                        G = zeros(size(F,1),3);
697                        G(end-8:end-6,:) = R;
698                        obj.Store_Covariance( F*Pa*F' + G*obj.Ql*G' );
699                    end
700                end
701            end
702            
703        end
